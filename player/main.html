function onPlayerReady(event) { event.target.mute(); // Τυχαία καθυστέρηση εκκίνησης let randomStartDelay = (MIN_START_DELAY + Math.random() * (MAX_START_DELAY - MIN_START_DELAY)) * 1000; setTimeout(() => { // Τυχαίο σημείο εκκίνησης μέσα στο πρώτο λεπτό let randomSeek = Math.floor(Math.random() * 60); event.target.seekTo(randomSeek, true); event.target.playVideo(); event.target.setPlaybackQuality('small'); console.log(`Player ξεκίνησε στο ID=${event.target.getVideoData().video_id}, από ${randomSeek}s`); scheduleNext(event.target); randomPause(event.target); }, randomStartDelay); } function randomPause(player) { // Τυχαία παύση μέσα στα πρώτα 5 λεπτά let pauseDelay = Math.random() * 300000; setTimeout(() => { player.pauseVideo(); console.log(`Player έκανε pause στο ID=${player.getVideoData().video_id}`); setTimeout(() => { player.playVideo(); console.log(`Player συνέχισε το ID=${player.getVideoData().video_id}`); }, 5000 + Math.random() * 5000); // pause 5–10s }, pauseDelay); }
